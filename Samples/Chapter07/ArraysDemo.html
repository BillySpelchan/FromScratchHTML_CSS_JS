Arrays

setting up

let names = ["Alice", "Bob", "Carole", "Eve",
    "Faith", "Grace", "Malory"];

for loop

let s = "";
for (let i = 0; i < names.length; ++i) {
    s += "names[" + i + "] = " + names[i] + "<br>";
}
document.getElementById("forloop").innerHTML=s;

for in

s = "";
for (let key in names) {
    s += `names[${key}]=${names[key]}
`;
}
document.getElementById("forin").innerHTML=s;

for of

s = "";
for (let value of names) {
    s += value + "
";
}
document.getElementById("forof").innerHTML=s;
